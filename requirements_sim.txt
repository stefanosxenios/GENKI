#
#   Official releases:  https://github.com/EPFL-LCSB/skimpy/releases
#
#   Option A — conda package (Linux / macOS, recommended):
#     conda config --add channels bioconda
#     conda config --add channels conda-forge
#     conda config --set channel_priority strict
#     conda install weilandtd::skimpy
#
#   Option B — from source (all platforms):
#     # First install system libraries (Ubuntu/Debian):
#     sudo apt install gfortran libsundials-dev libflint-dev libgmp-dev
#     # Then clone and install:
#     git clone https://github.com/EPFL-LCSB/skimpy.git /path/to/skimpy
#     pip install -e /path/to/skimpy
#
#   pytfa is installed automatically as a skimpy dependency; if not:
#     pip install pytfa
#
#   For commercial solver setup (CPLEX / GUROBI, required for pytfa sampling):
#     https://github.com/EPFL-LCSB/skimpy/blob/master/SOLVERS.md
#

#   pip install -r requirements_sim.txt
